/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<List<Integer>> levelOrderBottom(TreeNode root) {

        List<List<Integer>> traverse = new ArrayList<>();

        if (root == null)
            return traverse;

        Queue<TreeNode> Q = new LinkedList<>();
        Q.offer(root);
        
        while (!Q.isEmpty()) {
            int size = Q.size();
            List<Integer> answer = new ArrayList<>();
            for (int i = 0; i < size; i++) {
                TreeNode top = Q.poll();
                answer.add(top.val);
                if (top.left != null)
                    Q.offer(top.left);
                if (top.right != null)
                    Q.offer(top.right);
            }
            traverse.add(answer);
        }

        Collections.reverse(traverse);
        return traverse;
    }
}
